Tree find( Tree T, int key )
{
    if(!T) return NULL;
    else if(T->element==key) return T;
    else if(T->element>key) return find(T->left,key);
    else if(T->element<key) return find(T->right,key);
}
void inorder(Tree T)
{
    if(!T) return;
    if(T->left) inorder(T->left);
